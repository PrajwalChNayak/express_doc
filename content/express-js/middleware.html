What is Middleware?

Middleware functions are functions that have access to the request object (req), the response object (res), and the next middleware function in the application's request-response cycle.

Middleware Function Structure

middleware-structure.js 
function middleware(req, res, next) {
  // Middleware logic here
  console.log('Time:', Date.now());
  
  // Call next() to pass control to the next middleware
  next();
}

req (Request)

Contains information about the HTTP request (headers, parameters, body, etc.)

res (Response)

Used to send the HTTP response back to the client

next (Next)

Function to pass control to the next middleware in the stack

Types of Middleware

app-middleware.js 
const express = require('express');
const app = express();

// Application-level middleware with no mount path
// Executes for every request
app.use((req, res, next) => {
  console.log('Time:', Date.now());
  next();
});

// Application-level middleware with mount path
app.use('/user/:id', (req, res, next) => {
  console.log('Request Type:', req.method);
  next();
});

// Multiple middleware functions
app.get('/user/:id', 
  (req, res, next) => {
    console.log('ID:', req.params.id);
    next();
  },
  (req, res, next) => {
    res.send('User Info');
  }
);
router-middleware.js 
const express = require('express');
const router = express.Router();

// Router-level middleware
router.use((req, res, next) => {
  console.log('Router middleware executed');
  next();
});

// Router-level middleware for specific path
router.use('/users', (req, res, next) => {
  console.log('Users route accessed');
  next();
});

router.get('/users/:id', (req, res) => {
  res.json({ id: req.params.id });
});

// Mount the router
app.use('/api', router);

Middleware Execution Order

Important!

Middleware execution order matters! Middleware is executed in the order it's defined.

execution-order.js 
const express = require('express');
const app = express();

// 1. First middleware
app.use((req, res, next) => {
  console.log('Middleware 1');
  next();
});

// 2. Second middleware
app.use((req, res, next) => {
  console.log('Middleware 2');
  next();
});

// 3. Route handler
app.get('/', (req, res) => {
  console.log('Route handler');
  res.send('Hello World!');
});

// 4. This won't execute for GET / requests
app.use((req, res, next) => {
  console.log('Middleware 3 - After route');
  next();
});

// Output for GET /:
// Middleware 1
// Middleware 2
// Route handler

‚úÖ Best Practices

  • ‚Ä¢ Security middleware first (helmet, cors)
  • ‚Ä¢ Logging middleware early
  • ‚Ä¢ Body parsing before routes
  • ‚Ä¢ Authentication middleware before protected routes
  • ‚Ä¢ Error handling middleware last

‚ùå Common Mistakes

  • ‚Ä¢ Forgetting to call next()
  • ‚Ä¢ Wrong middleware order
  • ‚Ä¢ Sending response without ending
  • ‚Ä¢ Multiple response calls
  • ‚Ä¢ Not handling errors properly

Creating Custom Middleware

basic-middleware.js 
// Simple logging middleware
const loggerMiddleware = (req, res, next) => {
  const timestamp = new Date().toISOString();
  const method = req.method;
  const url = req.url;
  
  console.log(`[${timestamp}] ${method} ${url}`);
  next();
};

// Request time middleware
const requestTime = (req, res, next) => {
  req.requestTime = Date.now();
  next();
};

// Response time middleware
const responseTime = (req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    console.log(`Request took ${duration}ms`);
  });
  
  next();
};

// Use the middleware
app.use(loggerMiddleware);
app.use(requestTime);
app.use(responseTime);
auth-middleware.js 
const jwt = require('jsonwebtoken');

// JWT Authentication middleware
const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({ 
      error: 'Access token required' 
    });
  }

  jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
    if (err) {
      return res.status(403).json({ 
        error: 'Invalid or expired token' 
      });
    }
    
    req.user = user;
    next();
  });
};

// Role-based authorization middleware
const requireRole = (role) => {
  return (req, res, next) => {
    if (!req.user || req.user.role !== role) {
      return res.status(403).json({ 
        error: 'Insufficient permissions' 
      });
    }
    next();
  };
};

// Usage
app.get('/protected', authenticateToken, (req, res) => {
  res.json({ message: 'Protected route accessed', user: req.user });
});

app.get('/admin', authenticateToken, requireRole('admin'), (req, res) => {
  res.json({ message: 'Admin only route' });
});
validation-middleware.js 
const Joi = require('joi');

// Generic validation middleware
const validate = (schema, property = 'body') => {
  return (req, res, next) => {
    const { error } = schema.validate(req[property]);
    
    if (error) {
      const errorMessage = error.details
        .map(detail => detail.message)
        .join(', ');
        
      return res.status(400).json({
        error: 'Validation failed',
        details: errorMessage
      });
    }
    
    next();
  };
};

// User validation schema
const userSchema = Joi.object({
  name: Joi.string().min(2).max(50).required(),
  email: Joi.string().email().required(),
  age: Joi.number().integer().min(18).max(120),
  password: Joi.string().min(8).pattern(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/)
});

// Query parameter validation
const paginationSchema = Joi.object({
  page: Joi.number().integer().min(1).default(1),
  limit: Joi.number().integer().min(1).max(100).default(10)
});

// Usage
app.post('/users', validate(userSchema), (req, res) => {
  // Create user logic
  res.json({ message: 'User created successfully' });
});

app.get('/users', validate(paginationSchema, 'query'), (req, res) => {
  // Get users with pagination
  res.json({ users: [], pagination: req.query });
});
rate-limit-middleware.js 
// Simple rate limiting middleware
const rateLimiter = (windowMs = 15 * 60 * 1000, maxRequests = 100) => {
  const requests = new Map();
  
  return (req, res, next) => {
    const clientId = req.ip;
    const now = Date.now();
    const windowStart = now - windowMs;
    
    // Get existing requests for this client
    const clientRequests = requests.get(clientId) || [];
    
    // Filter out old requests
    const recentRequests = clientRequests.filter(time => time > windowStart);
    
    if (recentRequests.length >= maxRequests) {
      return res.status(429).json({
        error: 'Too many requests',
        message: `Rate limit exceeded. Try again later.`,
        retryAfter: Math.ceil(windowMs / 1000)
      });
    }
    
    // Add current request
    recentRequests.push(now);
    requests.set(clientId, recentRequests);
    
    next();
  };
};

// Advanced rate limiting with express-rate-limit
const rateLimit = require('express-rate-limit');

const createRateLimit = (windowMs, max, message) => {
  return rateLimit({
    windowMs,
    max,
    message: {
      error: 'Rate limit exceeded',
      message
    },
    standardHeaders: true,
    legacyHeaders: false
  });
};

// Different limits for different endpoints
const generalLimit = createRateLimit(
  15 * 60 * 1000, // 15 minutes
  100, // 100 requests per window
  'Too many requests from this IP'
);

const authLimit = createRateLimit(
  15 * 60 * 1000, // 15 minutes
  5, // 5 login attempts per window
  'Too many login attempts'
);

// Usage
app.use('/api/', generalLimit);
app.use('/auth/login', authLimit);

Middleware Best Practices

‚úÖ Do's

  • ‚Ä¢ Always call next() unless ending the request
  • ‚Ä¢ Handle errors properly
  • ‚Ä¢ Keep middleware functions focused and small
  • ‚Ä¢ Use async/await for asynchronous operations
  • ‚Ä¢ Order middleware logically
  • ‚Ä¢ Document your custom middleware
  • ‚Ä¢ Test middleware in isolation

‚ùå Don'ts

  • ‚Ä¢ Don't forget to call next()
  • ‚Ä¢ Don't call next() multiple times
  • ‚Ä¢ Don't send multiple responses
  • ‚Ä¢ Don't ignore error handling
  • ‚Ä¢ Don't make middleware too complex
  • ‚Ä¢ Don't hardcode values, use configuration
  • ‚Ä¢ Don't block the event loop

üéØ Key Takeaways

  • ‚Ä¢ Middleware executes in order of definition
  • ‚Ä¢ Each middleware can modify req and res objects
  • ‚Ä¢ Call next() to continue to the next middleware
  • ‚Ä¢ Use middleware for cross-cutting concerns
  • ‚Ä¢ Error-handling middleware has 4 parameters
  • ‚Ä¢ Built-in middleware covers common needs
